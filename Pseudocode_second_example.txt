An algorithm for searching newspapers based on specific time periods
READ "from" and "to"

		Print "De Aanval"
	ELSE
		Print "De Aabode"
	ELSE
		Print "Aambeeld"
	ELSE
		Print "Cereales vadenses"
	ELSE
		Print "Christofoor"
	ELSE

	ELSE
		Print "Eendracht maakt macht"

	ELSE

		ENDIF;
